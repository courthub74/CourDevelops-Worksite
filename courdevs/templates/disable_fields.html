Disable Fields By CheckBox


  1. Disable Javascript for OnSite when InStudio is checked
  2. 'Onsite' value is disabled when 'Instudio' is checked
    You can plug in ANY two values for this sequence
    <script type="text/javascript">
    $<function() {
    $("#checked").change(function() {
    var set = this.checked;
    if (set) {
    $("#other").prop("disabled", false);
    $("#other2").prop("disabled", false);
    $("#other3").prop("disabled", false);
    $("#other4").prop("disabled", false);
    $("#other5").prop("disabled", false);
    }
    else {
    $("#other").prop("disabled", true);
    }
    });
    });
    $(function() {
    $("#checked2").change(function() {
    var set2 = this.checked;
    if (set2) {
    $("#other").prop("disabled", true);
    $("#other2").prop("disabled", true);
    $("#other3").prop("disabled", true);
    $("#other4").prop("disabled", true);
    $("#other5").prop("disabled", true);
    }
    else {
    $("#other").prop("disabled", false);
    }
    });
    });
    </script>
