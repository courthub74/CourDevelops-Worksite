    # BRACES

    import re

    text = "Yo! You there? Yo!!! ... Yo!!!!!!"

    pattern = "Yo!{3,}"
    #ABOVE finds Yo with 3 exclamation points behind it

    regex = re.compile(pattern)

    all_matches = regex.findall(text)
    print(all_matches)
    # BRACKETS

    import re

    text = "123 city 456 %#$ loRS ..."

    pattern = "[aeiou]"
    #'[ ]' allow you to check for specific characters
    #ABOVE is checking for vowels
    #[aeiou][aeiou] checks to two consecutive vowels and so forth

    regex = re.compile(pattern)

    all_matches = regex.findall(text)
    print(all_matches)

    pattern = "\d"

    #'\d\d\d' matches triple and so forth
    #'\D' matches non digits same as numerical pattern

    regex = re.compile(pattern)

    all_matches = regex.findall(text)
    print(all_matches)
