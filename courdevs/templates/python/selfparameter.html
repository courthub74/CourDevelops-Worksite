SELF-Parameter

  1. # The self parameter is a reference to the current instance of the class

    # and is used to access variables that belong to the class

    class Player:
    def __init__(self, name, ppg):
    self.name = name
    self.ppg = ppg

    def printstat(a,b):
    print(str(a.name) + " Averages "
    + str(b.ppg) + " ppg" )

    # Make sure that the def is indented properly under the class
    or you will get an error stating:
    'Player' has no attribute 'printstat'
    # King James

    bron = Player("Lebron", 37)
    # ABOVE sets a variable 'bron' and
    passes two params thru the 'Player' class

    bron.printstat(bron)
    # ABOVE prints out the variable 'bron' through the 'printstat' function

    prints: Lebron Averages 37 ppg
    # Jimmy Butler

    jimmy = Player("Jimmy", 26)
    # ABOVE sets a variable 'jimmy' and
    passes two params thru the 'Player' class

    jimmy.printstat(jimmy)
    # ABOVE prints out the variable 'bron' through the 'printstat' function
